void ajusteHora()
{
    printf("\t\t CONFIGURAR HORA\n");
    printf("Digite a hora: ");
    scanf("%d",&hora);
    while(hora < 0 || hora > 23)
    {
        printf("Digite a hora de 0 a 23: ");
        scanf("%d",&hora);
    }
    printf("\nDigite os minutos: ");
    scanf("%d",&minutos);
    while(minutos < 0 || minutos > 59)
    {
        printf("\nDigite os minutos de 0 a 59: ");
        scanf("%d",&minutos);
    }
    printf("\nDigite o dia: ");
    scanf("%d",&dia);
    while(dia < 1 || dia > 30)
    {
        printf("\nDigite o dia de 1 a 30: ");
        scanf("%d",&dia);
    }
    printf("\nDigite o mes: ");
    scanf("%d",&mes);
    while(mes < 1 || mes > 12)
    {
        printf("\nDigite o mes de 1 a 12: ");
        scanf("%d",&mes);
    }
    printf("\nDigite o ano: ");
    scanf("%d",&ano);
    while(ano < 0)
    {
        printf("\nDigite o ano maior que 0: ");
        scanf("%d",&ano);
    }
    for(segundos = 0; segundos < 8; segundos++)
    {
        if(segundos == 60)
        {
            segundos = 0;
            minutos++;
        }
        if( minutos == 60)
        {
            minutos = 0;
            hora++;
        }
        if(hora == 24)
        {
            hora = 0;
            dia++;
        }
        if( dia == 30)
